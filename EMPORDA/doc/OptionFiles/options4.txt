#! Block entroy coder with reduced prediction mode and column oriented mode

# MAX BIT SIZE FOR SAMPLE 4 bits
DYNAMIC_RANGE=16

# ENCODING ORDER 1 = BSQ, 0 = BI
SAMPLE_ENCODING_ORDER=1

# SUBFRAME INTERLEAVING DEPTH USED IN BI ENCODING 16 bits
SUBFRAME_INTERLEAVING_DEPTH=0

# INTEGER NUMBER OF OUTPUT WORDS 3 bits
OUTPUT_WORD_SIZE=4

# ENTROPY CODER TYPE USED: 0 SAMPLE ADAPTIVE, 1 BLOCK ADAPTIVE
ENTROPY_CODER_TYPE=1

# NUMBER OF PREDICTION BANDS 4 bits
NUMBER_PREDICTION_BANDS=3

# 0 = FULL PREDICTION MODE, 1 = REDUCED PREDICTION MODE
PREDICTION_MODE=1

# 0 = NEIGHBOR ORIENTED SUM MODE, 1 = COLUMN ORIENTED SUM MODE
LOCAL_SUM_MODE=1

# SIZE OF THE REGISTER 6 bits
REGISTER_SIZE=32

# RESOLUTION OF WEIGHT COMPONENTS 4 bits
WEIGHT_COMPONENT_RESOLUTION=13

# WEIGHT UPDATE SCALING EXPONENT CHANGE INTERVAL 4 bits
WEIGHT_UPDATE_SECI=6

# WEIGHT UPDATE SCALING EXPONENT INITIAL PARAMETER 4 bits
WEIGHT_UPDATE_SE=-1

# WEIGHT UPDATE SCALING EXPONENT 4 bits
WEIGHT_UPDATE_SEFP=3

# 0 = DEFAULT INITIALIZATION, 1 = CUSTOM INITIALIZATION
WEIGHT_INITIALIZATION_METHOD=0

# WEIGHT INITIALIZATION TABLE FLAG, 0 = NOT INCLUDED IN METADATA, 1 = OTHERWISE
WEIGHT_INITIALIZATION_TF=0

# WEIGHT INITIALIZATION RESOLUTION 5 bits
WEIGHT_INITIALIZATION_RESOLUTION=0

# BLOCK SIZE, MUST BE 8 or 16
BLOCK_SIZE=16

# REFERENCE SAMPLE INTERVAL 12 bits
REFERENCE_SAMPLE_INTERVAL=128
